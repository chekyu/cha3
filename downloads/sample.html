  <script>

    const $ = (s, el=document) => el.querySelector(s);

    const $$ = (s, el=document) => Array.from(el.querySelectorAll(s));


    const backdrop = $("#backdrop");

    const openBtn = $("#openModalBtn");

    const input = $("#targetServiceInput");


    const checkAll = $("#checkAll");

    const items = $$(".item");

    const chipsEl = $("#chips");

    const hintEl = $("#hint");


    const resetBtn = $("#resetBtn");

    const applyBtn = $("#applyBtn");


    function openModal(){

      backdrop.style.display = "flex";

      backdrop.setAttribute("aria-hidden", "false");

      // 스크롤 잠금(선택)

      document.body.style.overflow = "hidden";

      renderSummary();

    }


    function closeModal(){

      backdrop.style.display = "none";

      backdrop.setAttribute("aria-hidden", "true");

      document.body.style.overflow = "";

    }


    function getSelected(){

      return items.filter(i => i.checked).map(i => i.value);

    }


    function syncCheckAll(){

      const selectedCount = getSelected().length;

      checkAll.checked = selectedCount > 0 && selectedCount === items.length;

      checkAll.indeterminate = selectedCount > 0 && selectedCount < items.length;

    }


    function renderSummary(){

      const selected = getSelected();


      // chips

      chipsEl.innerHTML = "";

      if(selected.length === 0){

        hintEl.textContent = "선택된 항목이 없습니다.";

        hintEl.style.display = "inline";

      }else{

        hintEl.textContent = "";

        hintEl.style.display = "none";


        selected.forEach(name => {

          const chip = document.createElement("span");

          chip.className = "chip";

          chip.innerHTML = `

            ${escapeHtml(name)}

            <button type="button" aria-label="${escapeHtml(name)} 삭제">×</button>

          `;

          chip.querySelector("button").addEventListener("click", () => {

            const target = items.find(i => i.value === name);

            if(target) target.checked = false;

            syncCheckAll();

            renderSummary();

          });

          chipsEl.appendChild(chip);

        });

      }


      syncCheckAll();

    }


    function escapeHtml(str){

      return String(str).replace(/[&<>"']/g, (m) => ({

        "&":"&amp;","<":"&lt;",">":"&gt;",'"':"&quot;","'":"&#39;"

      }[m]));

    }


    // Events

    openBtn.addEventListener("click", openModal);


    // 바깥 클릭 닫기

    backdrop.addEventListener("click", (e) => {

      if(e.target === backdrop) closeModal();

    });


    // ESC 닫기

    window.addEventListener("keydown", (e) => {

      if(e.key === "Escape" && backdrop.style.display === "flex"){

        closeModal();

      }

    });


    checkAll.addEventListener("change", () => {

      const checked = checkAll.checked;

      items.forEach(i => i.checked = checked);

      renderSummary();

    });


    items.forEach(i => i.addEventListener("change", renderSummary));


    resetBtn.addEventListener("click", () => {

      checkAll.checked = false;

      checkAll.indeterminate = false;

      items.forEach(i => i.checked = false);

      renderSummary();

    });


    applyBtn.addEventListener("click", () => {

      const selected = getSelected();

      // 외부 input에 반영(예시)

      input.value = selected.join(", ");

      closeModal();

    });


    // 초기값

    renderSummary();

  </script>
